洛谷的一套题，作为自己的算法模板保存起来吧OwO..

```cpp
#include<iostream>
using namespace std;
int a[1000];

int find(int k) {
	if (a[k] == k) return k;
	return a[k] = find(a[k]);
}

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 1; i <= n; i++) {
		a[i] = i;
	}
	while (m--) {
		int temp , x,y;
		cin >> temp >> x>>y;
		if (temp == 1) {
			a[find(x)] = find(y);
		}
		else {
			if (find(x) == find(y))
				cout << "Y" << endl;
			else
				cout << "N" << endl;
		}
	}
	return 0;
}
```
